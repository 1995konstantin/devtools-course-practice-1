#############################################
##### Testing
#############################################

enable_testing()

add_test(
    NAME can_Run
    COMMAND ${appname}
)

add_test(
    NAME can_Print_Help_Without_Arguments
    COMMAND ${appname}
)
set_tests_properties (can_Print_Help_Without_Arguments
  PROPERTIES PASS_REGULAR_EXPRESSION "This is a simple complex calculator.")

add_test(
    NAME can_Check_Number_Of_Arguments
    COMMAND ${appname} 1 2
)
set_tests_properties (can_Check_Number_Of_Arguments
  PROPERTIES PASS_REGULAR_EXPRESSION "ERROR: Should be 5 arguments.")

add_test(
    NAME can_Detect_Wrong_Number_Format
    COMMAND ${appname} 1 ip 2 3 +
)
set_tests_properties (can_Detect_Wrong_Number_Format
  PROPERTIES PASS_REGULAR_EXPRESSION "Wrong number format!")

add_test(
    NAME can_Detect_Wrong_Operation_Format
    COMMAND ${appname} 1 1 2 3 +trash
)
set_tests_properties (can_Detect_Wrong_Operation_Format
  PROPERTIES PASS_REGULAR_EXPRESSION "Wrong operation format!")

add_test(
    NAME can_Add_Complexs
    COMMAND ${appname} 2.5 4.5 1.0 4.5 +
)
set_tests_properties (can_Add_Complexs
  PROPERTIES PASS_REGULAR_EXPRESSION "Real = 3.5 Imaginary = 9")

add_test(
    NAME can_Diff_Complexs
    COMMAND ${appname} -17 0 15 -3.5 -
)
set_tests_properties (can_Diff_Complexs
  PROPERTIES PASS_REGULAR_EXPRESSION "Real = -32 Imaginary = 3.5")

add_test(
    NAME can_Mult_Complexs
    COMMAND ${appname} 13 -2 4.6 7.3 *
)
set_tests_properties (can_Mult_Complexs
  PROPERTIES PASS_REGULAR_EXPRESSION "Real = 74.4 Imaginary = 85.7")

add_test(
    NAME can_Divide_Complexs
    COMMAND ${appname} 27 30 15 20 /
)
set_tests_properties (can_Divide_Complexs
  PROPERTIES PASS_REGULAR_EXPRESSION "Real = 1.608 Imaginary = -0.144")

add_test(
    NAME can_Detect_Divide_By_Zero
    COMMAND ${appname} 27 30 0 0 /
)
set_tests_properties (can_Detect_Divide_By_Zero
  PROPERTIES PASS_REGULAR_EXPRESSION "Can't divide by zero")
